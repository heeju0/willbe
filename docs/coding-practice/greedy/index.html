T = int(input())
def flip(s, k):
idx = s.find('-')
s = list(s)
if idx != -1 and idx+k <= len(s):
s[idx:idx+k] = map(lambda x: '+' if x == '-' else '-',s[idx:idx+k])
return ''.join(s)
for t in range(T):
s, k = input().split()
k = int(k)
res = [s]
while True:
s = flip(s, k)
if s in res:
break
else:
res.append(s)
if '+'*len(s) in res:
break
print('Case #{}: {}'.format(t+1, len(res)-1 if '+'*len(s) in res else 'IMPOSSIBLE'))
