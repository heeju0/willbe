def flip(s, k):

for t in range(T):

k = int(k)
res = [s]
while True:
s = flip(s, k)

res.append(s)
if '+'*len(s) in res:
break
print('Case #{}: {}'.format(t+1, len(res)-1 if '+'*len(s) in res else 'IMPOSSIBLE'))
