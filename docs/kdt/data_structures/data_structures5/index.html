def solution(tickets):
routes = {}
for t in tickets:
routes[t[0]] = routes.get(t[0], []) + [t[1]]
# O(n log n)
for r in routes:
routes[r].sort(reverse=True)
stack = ['ICN']
path = []
# O(n)
while stack:
top = stack[-1]
if top not in routes or len(routes[top]) == 0:
path.append(stack.pop())
else:
stack.append(routes[top].pop())
return path[::-1]
