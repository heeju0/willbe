def solution(numbers):
# O(n)
numbers = [str(x) for x in numbers]
# O(n log n)
numbers.sort(key=lambda x: (x*4)[:4], reverse=True)
if numbers[0] == '0':
# O(1)
answer = '0'
else:
# O(n)
answer = ''.join(numbers)
return answer
